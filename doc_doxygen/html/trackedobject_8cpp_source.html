144 bool TrackedObject::change_trajectory_section(int64_t oldTimestamp, int64_t newTimestamp, Selection position, unsigned long timePosition, unsigned long frameNumber)
145 {
146  // The oldTimestamp needs to exist
147  assert(trajectorySections.find(oldTimestamp) != trajectorySections.end());
148 
149  if (endTimestampSet && endTimestamp < newTimestamp)
150  return false; // Section cannot begin after the end of the tracking period
151 
152  // currentSection will be unset, therefore trackingAlgorithm deletion is needed
153  if (currentSection && currentSection->trackingAlgorithm)
154  { // if currentSection is set and has initialized trackingAlgorithm, delete the trackingAlgorithm
155  // as it is not needed anymore; if needed, it would get initialized again
156 
157  trajectory.erase(trajectory.find(currentSection->initialTimestamp), trajectory.end()); // Clear all the trajectory from Current section initial timestamp till the end
158  delete currentSection->trackingAlgorithm;
159  currentSection->trackingAlgorithm = nullptr;
160  }
161 
162  currentSection = nullptr; // let track_next() to find correct currentSection and nextSection
163  nextSection = false;
164  allProcessed = false;
165 
166  auto oldSection = trajectorySections.find(oldTimestamp);
167 
168  if (oldSection == trajectorySections.begin())
169  { // The Beginning section is being changed
170 
171  assert(initialTimestamp == oldTimestamp);
172 
173  trajectory.clear(); // All trajectory will be counted from the beginning
174 
175  // Delete all trajectory changes (sections) that occur before this section as this is the beginning.
176  // This situation happens when user moves the Beginning forward while some Section was
177  // defined on some position between the old Beginning and the new one.
178  trajectorySections.erase(trajectorySections.begin(), trajectorySections.find(newTimestamp));
179 
180  initialTimestamp = newTimestamp;
181 
182  }
183  else
184  { // Trajectory change section is being change, not Beginning
185  trajectory.erase(trajectory.find(newTimestamp), trajectory.end()); // Clear all the trajectory from the timestamp at which the section begins till the end
186

188  initialTimestamp = newTimestamp;

191  { // Section is updated to begin later
192 
193  // V tomhle pripade je treba, aby bylo z trajectory odstraneno vsechno pocinaje
194  // zacatku PREDCHOZI sekce od oldTimestamp. To z toho duvodu, ze od oldTimestamp

196  // oldTimestamp nezacina (ta soucasna se meni ne currentTimestamp), tak je treba pozici

199  // zacatku teto predchozi sekce.

223 bool TrackedObject::delete_trajectory_section(int64_t timestamp)
224 {
225  if (timestamp == initialTimestamp) // Beginning section cannot be deleted
226  {
227  qDebug() << "Beginning section cannot be deleted";
228  return false;
229  }
230 
231  if (currentSection && (currentSection->initialTimestamp >= timestamp))
232  { // This trajectorySection will be erased below(==timestamp) or
233  // takes place after the section being deleted(>timestamp), therefore needs to be unset
234  if (currentSection->trackingAlgorithm)
235  {
236  delete currentSection->trackingAlgorithm;
237  currentSection->trackingAlgorithm = nullptr;
238  }
239 
240  currentSection = nullptr;
241  nextSection = false;
242  }
243  else if (nextSection && nextSectionTimestamp == timestamp)
244  { // The deleted section should have been the next section.
245  // Set the following section or false if no later section exists.
246 
247  auto nextSectionIterator = trajectorySections.upper_bound(timestamp);
248  if (nextSectionIterator != trajectorySections.end())
249  {
250  nextSection = true;
251  nextSectionTimestamp = nextSectionIterator->first;
252  }
253  else
254  nextSection = false;
255  }
256  allProcessed = false;
257 
258  // previousSection always exists as Beginning section cannot be deleted.
259  auto previousSection = --(trajectorySections.find(timestamp));
260  trajectorySections.erase(timestamp);
261 
262  // Remove all entries that exist for frames at timestamps higher than the beginning
263  // of the PREVIOUS section. It is needed so the trackingAlgorithm from the previous section
264  // goes from its beginning to have the correct data. Thereby, the deleted trajectory will be recounted.
265  trajectory.erase(trajectory.find(previousSection->first), trajectory.end());
266

356 Selection TrackedObject::track_next(VideoFrame const *frame)
357 {
358  if (!currentSection || (nextSection && frame->get_timestamp() >= nextSectionTimestamp))
359  { // Enters new section
360  // Close old section and initialize a new one.
361 
362  int64_t currentSectionTimestamp;
363  if (!currentSection)
364  { // First section or change to trajectorySections was made
365  if (trajectory.begin() == trajectory.end())
366  { // Nothing is processed, set Beginning section
367  auto trajectoryIterator = trajectorySections.begin();
368  currentSection = &(trajectoryIterator->second);
369 
370  currentSectionTimestamp = trajectoryIterator->first; // Is later used to count the nextSectionTimestamp
371  }
372  else
373  {
374  int64_t lastProcessed = trajectory.rbegin()->first;
375 
376  // Find section that begins after the last processed frame.
377  // That should be the next frame. If not - entries would be missing
378  // and that would cause an error.
379  auto trajectoryIterator = trajectorySections.upper_bound(lastProcessed);
380  currentSection = &(trajectoryIterator->second);
381  currentSectionTimestamp = trajectoryIterator->first;
382 
383  }
384  }
385  else
386  { // currentSection was set
387  // Entering next section
388  if (currentSection->trackingAlgorithm)
389  { // delete the trackingAlgorithm as it is not needed anymore;
390  // if needed, it would get initialized again
391  delete currentSection->trackingAlgorithm;
392  currentSection->trackingAlgorithm = nullptr;
393  }
394  currentSection = &(trajectorySections[nextSectionTimestamp]);
395  currentSectionTimestamp = nextSectionTimestamp;
396  }
397 
398  auto nextSectionIterator = trajectorySections.upper_bound(currentSectionTimestamp);
399  if (nextSectionIterator != trajectorySections.end())
400  {
401  nextSection = true;
402  nextSectionTimestamp = nextSectionIterator->first;
403  }
404  else
405  nextSection = false;
406 
407  // Now initialize the new section's tracking algorithm
408  Selection centerizedPosition;
409 
410 
411  currentSection->trackingAlgorithm = new TrackingAlgorithm(*(frame->get_mat_frame()),
412  currentSection->initialPosition, centerizedPosition);
413 
414  trajectory[frame->get_timestamp()] = TrajectoryEntry(centerizedPosition, frame->get_time_position(), frame->get_frame_number());
415 
416  return centerizedPosition;
417  }
418 
419  Selection result = currentSection->trackingAlgorithm->track_next_frame(*(frame->get_mat_frame()));
420 
421  trajectory[frame->get_timestamp()] = TrajectoryEntry(result, frame->get_time_position(), frame->get_frame_number());
422 
423 
424  int64_t lastProcessedTimestamp;
425  if (endTimestampSet && get_last_processed_timestamp(lastProcessedTimestamp) && lastProcessedTimestamp == endTimestamp)
