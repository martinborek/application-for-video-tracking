407 {

409 
410  int frameFinished;
411  bool lastFrames = false;
412 
413  while (true)
414  {
415  if (av_read_frame(formatContext, &packet) < 0)
416  {
417  av_free_packet(&packet);
418 
419  if (videoContext->codec->capabilities & CODEC_CAP_DELAY)
420  { // delayed frames at the end of the video
421  lastFrames = true;
422  qDebug() << "last packets";
423  packet.data = NULL;
424  packet.size = 0;
425  packet.stream_index = videoStreamID;
426  }
427  else
428  return false;
429  }
430 
431 
432 
433  // Is this a packet from the video stream?
434  if (packet.stream_index == videoStreamID)
435  {
436  //qDebug() << "readFrame pts" <<newFrame->pts;
437 
438  // Decode video frame
439  if (avcodec_decode_video2(videoContext, newFrame, &frameFinished, &packet) < 0)
440  {
441  av_free_packet(&packet);
442  return false;
443  }
444 
445 
446  // frameFinished==0 -> no frame could be decompressed
447  if (frameFinished)
448  {
449  lastTimestamp = newFrame->pkt_pts;
450  av_free_packet(&packet);
451 
452  if (!firstTimestampSet)
453  {
454  firstTimestamp = lastTimestamp;
455  qDebug() << "First timestamp is" << firstTimestamp;
456  firstTimestampSet = true;
457  }
458 
459  if (!onlyVideoPackets)
460  *isAudio = false;
461 
462  return true;
463  }
464  else if (lastFrames)
465  {
466  qDebug() << "last frames finished";
467  return false;
468  }
469  }
470  else if(!onlyVideoPackets && packet.stream_index == audioStreamID)
471  { // Do not forget to free packet after using it
472  *isAudio = true;
473  return true;
474 
475  }
476  // Free the packet that was allocated by av_read_frame
477  av_free_packet(&packet);
478  }

480  return false;
