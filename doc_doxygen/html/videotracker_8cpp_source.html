206 bool VideoTracker::track_frame(VideoFrame const *originalFrame, cv::Mat &result, QProgressDialog *progressDialog, bool previousTimestampSet,
207  int64_t previousTimestamp)
208 {
209  int64_t currentTimestamp = originalFrame->get_timestamp();
210 
211  if (!trackedObjects.empty())
212  {
213  Selection trackedPosition;
214  //for (TrackedObject *object: trackedObjects)
215  for (auto object: trackedObjects)
216  {
217  if ((object->get_initial_timestamp() > currentTimestamp) ||
218  ((object->is_end_timestamp_set() && object->get_end_timestamp() < currentTimestamp)))
219  continue; // In this case current timestamp is out of the range of this tracked object
220 
221  int64_t lastProcessedTimestamp;
222  bool lastProcessedTimestampSet = object->get_last_processed_timestamp(lastProcessedTimestamp); // If the frame was already processed but the whole video is not yet processed
223  if (object->is_all_processed() || (lastProcessedTimestampSet && currentTimestamp <= lastProcessedTimestamp))
224  {
225  if (!object->get_position(currentTimestamp, trackedPosition))// Position is already known as this frame has already been processed.
226  {
227  return false;
228  }
229  }
230  else if (previousTimestampSet && (lastProcessedTimestamp == previousTimestamp)) // Is used only with function "get_next_frame"; instead of seeking frame it uses the current one as it is the right one when obtained by get_next_frame(); that's much faster
231  trackedPosition = object->track_next(originalFrame);
237  else
238  {
239  if (progressDialog)
240  {
241  progressDialog->show();
242  }
243  if (lastProcessedTimestampSet)
244  {
245  // Set to the last processed position. This frame won't be used, but allows to use "get_next_frame".
246  // It might be possible to skip this frame and find right the desired (next) one.
247  // However, it would be more complicated and won't make it much faster.
248  if (!player->get_frame_by_timestamp(tempFrame, lastProcessedTimestamp))
249  return false;
250 
251  if (!player->get_next_frame(tempFrame)) // No more frames => all processed for this object
252  {
253  object->set_all_processed(true);
254  continue;
255  }
256  }
257  else
258  { // So far no frame is processed, therefore lastProcessedTimestamp is not set
259  if (!player->get_frame_by_timestamp(tempFrame, object->get_initial_timestamp()))
260  return false;
261  }
262 
263  while (true)
264  {
265  qApp->processEvents(); // Keeps progress bar active
266  trackedPosition = object->track_next(tempFrame);
267 
268  if (tempFrame->get_timestamp() >= currentTimestamp)
269  break;
270 
271  if (progressDialog)
272  {
273  if (progressDialog->wasCanceled()) // User cancelled the progress dialog
274  throw UserCanceledException();
275  }
276 
277  qApp->processEvents(); // Keeps progress bar active
278  if (!player->get_next_frame(tempFrame))
279  return false; // Could not reach desired frame
280  }
281 
282  //while (object->get_last_processed_timestamp() < currentTimestamp);
283 
284  assert(currentTimestamp == tempFrame->get_timestamp());
285 
286 
287  if (progressDialog)
288  {
289  if (progressDialog->wasCanceled()) // User cancelled the progress dialog
290  throw UserCanceledException();
291  }
292  }
293 
294  if (!object->draw_mark(result, *originalFrame->get_mat_frame(), currentTimestamp))
295  return false;
296 
297  }
298  //progressDialog->reset();
299  return true;
300  }
301 
302  return true;
303 }

305 bool VideoTracker::track_object(unsigned int objectID, QProgressDialog *progressDialog)
306 {
307  auto object = trackedObjects[objectID];
308  if (object->is_all_processed())
309  {
310  return true; // This object is processed throughout all its range
311  }
312  else
313  {
314  int64_t endTimestamp = object->get_end_timestamp();
315 
316  int64_t lastProcessedTimestamp;
317  bool lastProcessedTimestampSet = object->get_last_processed_timestamp(lastProcessedTimestamp); // If the frame was already processed but the whole video is not yet processed
318 
319  if (lastProcessedTimestampSet)
320  {
321  // Set to the last processed position. This frame won't be used, but allows to use "get_next_frame".
322  // It might be possible to skip this frame and find right the desired (next) one.
323  // However, it would be more complicated and won't make it much faster.
324  if (!player->get_frame_by_timestamp(tempFrame, lastProcessedTimestamp))
325  {
326  //progressDialog->reset();
327  return false;
328  }
329 
330  if (!player->get_next_frame(tempFrame)) // No more frames => all processed for this object
331  {
332  object->set_all_processed(true);
333  return true;
334  }
335  }
336  else
337  { // So far no frame is processed, therefore lastProcessedTimestamp is not set
338  if (!player->get_frame_by_timestamp(tempFrame, object->get_initial_timestamp()))
339  {
340  //progressDialog->reset();
341  return false;
342  }
343  }
344 
345  while (true)
346  {
347  qApp->processEvents(); // Keeps progress bar active
348  object->track_next(tempFrame); // Also sets object->allProcessed
349 
350  if (object->is_end_timestamp_set() && (tempFrame->get_timestamp() >= endTimestamp))
351  break;
352 
353  if (progressDialog->wasCanceled()) // User canceled the progress dialog
354  {
355  // reads last tracked frame; is used for knowing where aborted
356  player->get_current_frame(currentFrame);
357  throw UserCanceledException();
358  }
359 
360  qApp->processEvents(); // Keeps progress bar active
361 
362  if (!player->get_next_frame(tempFrame))
363  {
364  if (object->is_end_timestamp_set())
365  {
366  qDebug() << "Warning-track_all(): Cannot read more frames but object->endTimestamp is higher";
367  }
368  object->set_all_processed(true);
369  break;
370  }
371  }
372  }
373  return true;
